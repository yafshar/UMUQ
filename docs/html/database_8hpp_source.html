603 bool database::resetSelection(int const minLength, int const maxLength)
604 {
605  if (minLength < 1 || maxLength < 1 || maxLength < minLength)
606  {
607  UMUQFAILRETURN("Wrong chain length size in the input!");
608  }
609 
610  if (!resetList())
611  {
612  UMUQFAILRETURN("Failed to allocate list array!");
613  }
614 
615  if (list)
616  {
617  // find out how many leaders we have
618  int nLeaders(0);
619  std::for_each(nSelection.begin(), nSelection.end(), [&](int const S_i) { nLeaders += static_cast<int>(S_i > 0); });
620 
621  {
622  // Get the pointer
623  int *nSel = nSelection.data();
624  // set the counter to 0
625  int idx(0);
626  // Fill the list
627  std::for_each(list.get(), list.get() + nSamplePoints, [&](sortType &l_i) {l_i.nSel = *nSel++; l_i.idx = idx++; });
628  }
629 
630  if (this->sort())
631  {
632  int nChains(nLeaders);
633 
634  // Breaking long chains
635  for (int i = 0; i < nLeaders; i++)
636  {
637  if (list[i].nSel > maxLength)
638  {
639  while (list[i].nSel > maxLength)
640  {
641  list[nChains].nSel = maxLength;
642  list[nChains].idx = list[i].idx;
643  list[i].nSel -= maxLength;
644  nChains++;
645  }
646  }
647  }
648 
649  idxPosition = static_cast<std::size_t>(nChains);
650 
651  // If the maximum chain length is 1, we do not need to sort it
652  if (maxLength != 1)
653  {
654  if (!this->sort())
655  {
656  UMUQFAILRETURN("Failed to sort list array!");
657  }
658  }
659 
660  if (minLength != 1)
661  {
662  // Increasing the chain size
663  for (int i = 0; i < nChains; i++)
664  {
665  if (list[i].nSel && list[i].nSel < minLength)
666  {
667  list[i].nSel = minLength;
668  }
669  }
670 
671  if (!this->sort())
672  {
673  UMUQFAILRETURN("Failed to sort list array!");
674  }
675  }
676 
677  return true;
678  }
679  UMUQFAILRETURN("Failed to sort list array!");
680  }
681  UMUQFAILRETURN("List is not set yet!");
682 }

716 bool database::updateWorkload(int const nChains)
717 {
718  if (nChains <= nSamplePoints)
719  {
720  // Greedy partitioning based on the workload
721  int const nWorkers = torc_num_workers();
722 
723  // Dummy work load array
724  std::vector<int> workLoad(nWorkers, 0);
725 
726  for (int i = 0; i < nChains; i++)
727  {
728  // Find the index of the work with minimum work load
729  int const minLoadWorker = static_cast<int>(std::distance(workLoad.begin(), std::min_element(workLoad.begin(), workLoad.end())));
730 
731  // Chain i would be assigned to that queue
732  queue[i] = minLoadWorker;
733 
734  // Sum up the new load on the current work load
735  workLoad[minLoadWorker] += nSelection[i];
736  }
737 
738  return true;
739  }
740  UMUQFAILRETURN("Wrong number of leading chains!");
741 }
