133 {
134  DataType fl = f0 + zl * (fp0 + zl * (f1 - f0 - fp0));

317 {
318  // Map [a,b] to [0,1]
319  DataType zmin = (xmin - a) / (b - a);
320  DataType zmax = (xmax - a) / (b - a);
321 
322  if (zmin > zmax)
