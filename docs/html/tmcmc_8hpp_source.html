601 bool tmcmc<FunctionType>::iterateInternal()
602 {
603  // Current Generation number
604  workInformation[0] = runData.currentGeneration;
605 
606  // Number of function value (from evaluation) at each sampling point
607  int nFvalue = 1;
608 
609  // Total number of sampling chains
610  int const nChains = leadersData.size();
611 
612  // Get the pointer to the sample points
613  double *leadersSamplePoints = leadersData.samplePoints.data();
614 
615  // Dimension of sample points
616  int const nDimSamplePoints = leadersData.nDimSamplePoints;
617 
618  // Get this generation probability
619  double *PJ = runData.generationProbabilty.data() + runData.currentGeneration;
620 
621  // Get the scaled chain covariance as \f$ \beta covariance \f$
622  if (!Data.useLocalCovariance)
623  {
624  std::transform(runData.covariance.begin(), runData.covariance.end(), localCovariance.begin(), [&](double const C) { return Data.bbeta * C; });
625  }
626 
627  // Number of burning steps
628  int const nBurningSteps = 0;
629 
630  // Loop through all the population size
631  for (int i = 0; i < nChains; i++)
632  {
633  // Sample number
634  workInformation[1] = i;
635 
636  // Step number, this is the first step
637  workInformation[2] = leadersData.nSelection[i];
638 
639  // Fill the input sample points from the leaders
640  std::copy(leadersSamplePoints, leadersSamplePoints + nDimSamplePoints, samplePoints.data());
641  leadersSamplePoints += nDimSamplePoints;
642 
643  if (Data.useLocalCovariance)
644  {
645  }
646 
647  // Create tasks
648  torc_create(-1, (void (*)())tmcmcMainTask<FunctionType>, 9,
649  1, MPIDatatype<long long>, CALL_BY_REF,
650  Data.nDim, MPIDatatype<double>, CALL_BY_COP,
651  1, MPIDatatype<int>, CALL_BY_COP,
652  1, MPIDatatype<double>, CALL_BY_RES,
653  1, MPIDatatype<int>, CALL_BY_COP,

656  Data.nDim * Data.nDim, MPIDatatype<double>, CALL_BY_COP,
657  1, MPIDatatype<int>, CALL_BY_COP,
658  reinterpret_cast<long long>(this), samplePoints.data(),
659  &Data.nDim, fValue.data() + i, &nFvalue, workInformation,

661  }
662

664  torc_waitall();
665  torc_disable_stealing();
666

668  fc.count();
669

672

901 template <class FunctionType>
902 void tmcmcMainTask(long long const TMCMCObj, double const *SamplePoints, int const *nSamplePoints,
903  double *Fvalue, int const *nFvalue, int const *WorkInformation,
904  double const *PJ, double const *Covariance, int const *nBurningSteps)
905 {
906  auto tmcmcObj = reinterpret_cast<tmcmc<FunctionType> *>(TMCMCObj);
907 
908  // If we have set the fitting function
909  if (!tmcmcObj->fitfun)
910  {
911  UMUQFAIL("Fitting function is not assigned! \n Fitting function must be set before initializing the TMCMC object!");
912  }
913 
914  int const nSamples = *nSamplePoints;
915  int const nFunvals = *nFvalue;
916  int const nSteps = WorkInformation[2];
917  int const nBurnSteps = *nBurningSteps;
918  double const pj = *PJ;
919 
920  // Create an array of sample points & function values for leaders
921  std::vector<double> leaderSamplePoints{SamplePoints, SamplePoints + nSamples};
922  std::vector<double> leaderFvalue{Fvalue, Fvalue + nFunvals};
923 
924  // Create an array of sample points & function values for candidates
925  std::vector<double> candidateSamplePoints(nSamples);
926  std::vector<double> candidateFvalue(nFunvals);
927 
928  return;
929 
930  // Map the data to the Eigen vector format without memory copy
931  umuq::EVectorMapType<double> EcandidateSamplePoints(candidateSamplePoints.data(), nSamples);
932 
933  // Create an array of work information
934  std::vector<int> workInformation{WorkInformation, WorkInformation + 3};
935 
936  // Sample mean for the first step
937  std::vector<double> chainMean{SamplePoints, SamplePoints + nSamples};
938 
939  // Create an instance of the pseudo random number generator
940  psrandom prng;
941 
942  for (int step = 0; step < nSteps + nBurnSteps; step++)
943  {
944  if (step > 0)
945  {
946  std::copy(leaderSamplePoints.begin(), leaderSamplePoints.end(), chainMean.begin());
947  }
948 
949  umuq::randomdist::multivariateNormalDistribution<double> mvnormal(chainMean.data(), Covariance, nSamples);
950 
951  // Generate a candidate
952  EcandidateSamplePoints = mvnormal.dist();
953 
954  // Check the candidate

956  for (int i = 0; i < nSamples; i++)
