544 void broadcastTask(long long const other)
545 {
546 #if HAVE_MPI == 1
547  auto obj = reinterpret_cast<runinfo *>(other);
548 
549  int const nDim(obj->nDim * obj->nDim);
550  int const maxGenerations(obj->maxGenerations);
551 
552  MPI_Request request[3];
553 
554  MPI_Ibcast(obj->covariance.data(), nDim, MPIDatatype<double>, 0, MPI_COMM_WORLD, &request[0]);
555  MPI_Ibcast(obj->generationProbabilty.data(), maxGenerations, MPIDatatype<double>, 0, MPI_COMM_WORLD, &request[1]);
556  MPI_Ibcast(&obj->currentGeneration, 1, MPI_INT, 0, MPI_COMM_WORLD, &request[2]);
557 
558  MPI_Waitall(3, request, MPI_STATUSES_IGNORE);
559 #endif // MPI
