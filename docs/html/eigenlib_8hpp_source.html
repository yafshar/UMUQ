345 {

385 template <typename DataType>
386 void forceSelfAdjointMatrixPositiveDefinite(DataType *dataPtr, int const nDim)
387 {
388  // First Map the data to Eigen Matrix format. (Symmetric Matrix)
389  EMapType<DataType, Eigen::ColMajor> eMatrix(dataPtr, nDim, nDim);
390 
391  // Fixed value to increase
392  DataType const increaseRate(0.01);
393 
394  // Fixed value to multiply by
395  DataType const fixedRate(1.01);
396 
397 #ifdef DEBUG
398  std::size_t iter(0);
399  if (nDim < 20)
400  {
401  UMUQMSG("eMatrix=", eMatrix);
402  }
403 #endif
404 
405  // Vector for diagonal elements
406  auto D = eMatrix.diagonal();
407 
408  // Whether we have negative or zero ( < machinePrecision) on the diagonal elements fo the matrix
409  bool const belowMachinePrecision = (D.array() <= machinePrecision<DataType>).any();
410 
411  if (belowMachinePrecision)
412  {
413  // Find the maximum absolute element on the diagonal of the matrix
414  auto MaxAbsD = D.cwiseAbs().maxCoeff();
415 
416  // Find the minimum element
417  auto const MinD = D.minCoeff();
418 
419  MaxAbsD *= increaseRate;
420 
421  // There is a negative component on the diagonal, and we should get rid of it.
422  if (MinD < 0)
423  {
424  MaxAbsD -= MinD;
425  }
426 
427  if (MaxAbsD < machinePrecision<DataType>)
428  {
429  MaxAbsD = machinePrecision<DataType>;
430  }
431 
432  for (int i = 0; i < nDim; i++)
433  {
434  eMatrix(i, i) += MaxAbsD;
435  }
436  // Now none of the diagonal elements are below machine precision
437  }
438  else
439  {
440  for (int i = 0; i < nDim; i++)
441  {
442  eMatrix(i, i) *= fixedRate;
443  }
444  }
445 
446  while (!isSelfAdjointMatrixPositiveDefinite<DataType>(dataPtr, nDim))
447  {
448 #ifdef DEBUG
449  iter++;
450  UMUQMSG("Iteration number ", iter, " to force the Covariance Matrix Positive Definite");
451  if (nDim < 20)
452  {
453  UMUQMSG("eMatrix=", eMatrix);
454  }
455 #endif
456  for (int i = 0; i < nDim; i++)
457  {
458  eMatrix(i, i) *= fixedRate;
459  }
460  }
461  return;
462 }

566 template <typename DataType, typename OutputDataType = double, VectorwiseOperation Direction = VectorwiseOperation::ColWise>
567 EMatrixX<OutputDataType> squaredL2Distance(EMatrixX<DataType> const &eMatrix)
568 {
569  switch (Direction)
570  {
571  case VectorwiseOperation::ColWise:
572  {
573  auto const nCols = eMatrix.cols();
574  EMatrixX<OutputDataType> columnsSquaredDistanceMatrix(nCols, nCols);
575  ERowVectorX<OutputDataType> vecC = eMatrix.array().square().colwise().sum().template cast<OutputDataType>();
576  columnsSquaredDistanceMatrix.rowwise() = vecC;
577  columnsSquaredDistanceMatrix.colwise() += vecC.transpose();
578  columnsSquaredDistanceMatrix -= 2 * eMatrix.transpose().template cast<OutputDataType>() * eMatrix.template cast<OutputDataType>();
579  return columnsSquaredDistanceMatrix;
580  }
581  break;
582  case VectorwiseOperation::RowWise:
583  {
584  auto const nRows = eMatrix.rows();
585  EMatrixX<OutputDataType> rowsSquaredDistanceMatrix(nRows, nRows);
586  EVectorX<OutputDataType> vecR = eMatrix.array().square().rowwise().sum().template cast<OutputDataType>();
587  rowsSquaredDistanceMatrix.colwise() = vecR;
588  rowsSquaredDistanceMatrix.rowwise() += vecR.transpose();
589  rowsSquaredDistanceMatrix -= 2 * eMatrix.template cast<OutputDataType>() * eMatrix.transpose().template cast<OutputDataType>();
590  return rowsSquaredDistanceMatrix;
591  }
592  break;
593  default:
594  UMUQFAIL("Unknown direction!");
595  break;
596  };
603 }
