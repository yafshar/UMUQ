171 template <typename DataType, class PolynomialType>
172 bool linearRegression<DataType, PolynomialType>::computeWeights(DataType *idata, DataType *iFvalue, int const nPoints)
173 {
174  if (nPoints < 1 || nPoints < minPointsRequired())
175  {
176  UMUQFAILRETURN("Number of input data points =", nPoints, " are not enough!");
177  }
178 
179  // Create an instance of a polynomial object with polynomial order
180  PolynomialType poly(nDim, polynomialOrder);
181 
182  // Get the monomials size
183  linearRegressionMonomialSize = poly.monomialsize();
184 
185  if (linearRegressionMonomialSize > linearRegressionkernelSize)
186  {
187  linearRegressionkernelSize = linearRegressionMonomialSize;
188  try
189  {
190  // Make sure of the correct kernel size
191  linearRegressionkernel.resize(linearRegressionkernelSize);
192  }
193  catch (...)
194  {
195  UMUQFAILRETURN("Failed to allocate memory!");
196  }
197  }
198  else
199  {
200  linearRegressionkernelSize = linearRegressionMonomialSize;
201  }
202 
203  // Right hand side vector
204  EVectorMapType<DataType> BV(iFvalue, nPoints);
205 
206  // Matrix A
207  EMatrixX<DataType> AM(nPoints, linearRegressionMonomialSize);
208 
209  // Solution vector
210  DataType *sv = linearRegressionkernel.data();
211  EVectorMapType<DataType> SV(sv, linearRegressionMonomialSize);
212 
213  // dummy array of data
214  std::unique_ptr<DataType[]> rowData(new DataType[linearRegressionMonomialSize]);
215  DataType *rowdata = rowData.get();
216 
217  // Loop over all query points
218  for (int i = 0; i < nPoints; i++)
219  {
220  // Index in idata array
221  std::ptrdiff_t const IdI = i * nDim;
222 
223  // Evaluates a monomial at a point \f$ {\mathbf x} \f$
224  poly.monomialValue(idata + IdI, rowdata);
225 
226  // Loop through the neighbors
227  for (int j = 0; j < linearRegressionMonomialSize; j++)
228  {
229  AM(i, j) = rowData[j];
230  }
231 
232  } // Loop over all points
233 
234  {
235  /*
236  * Two-sided Jacobi SVD decomposition, ensuring optimal reliability and accuracy.
237  * Thin U and V are all we need for (least squares) solving.
238  */
239  Eigen::JacobiSVD<EMatrixX<DataType>> svd(AM, Eigen::DecompositionOptions::ComputeThinU | Eigen::DecompositionOptions::ComputeThinV);
240 
241  /*
242  * SV contains the least-squares solution of
243  * \f$ {\mathbf A} ({\mathbf X}) ={\mathbf b}, \f$
244  * using the current SVD decomposition of A.
245  */
246  SV = svd.solve(BV);
247  }
248 
249  return true;
250 }

252 template <typename DataType, class PolynomialType>
253 bool linearRegression<DataType, PolynomialType>::solve(DataType *qdata, DataType *qFvalue, int const nqPoints)
254 {
255  // Create an instance of a polynomial object with polynomial order
256  PolynomialType poly(nDim, polynomialOrder);
257 
258  if (poly.monomialsize() != linearRegressionkernelSize)
259  {
260  UMUQFAILRETURN("Polynomial order has changed between Linear regression construction and its solution!");
261  }
262 
263  if (qFvalue == nullptr)
264  {
265  try
266  {
267  qFvalue = new DataType[nqPoints];
268  }
269  catch (std::bad_alloc &e)
270  {
271  UMUQFAILRETURN("Failed to allocate memory!");
272  }
273  }
274 
275  // Get the pointer to the kernel weights
276  DataType *sv = linearRegressionkernel.data();
277 
278  // dummy array of data
279  std::unique_ptr<DataType[]> rowData(new DataType[linearRegressionMonomialSize]);
280  DataType *rowdata;
281 
282  // Loop over all query points
283  for (int i = 0; i < nqPoints; i++)
284  {
285  // Index in qdata array
286  std::ptrdiff_t const IdI = i * nDim;
287 
288  rowdata = rowData.get();
289 
290  // Evaluates a monomial at a point \f$ {\mathbf x} \f$
291  poly.monomialValue(qdata + IdI, rowdata);
292 
293  DataType sum(0);
294  std::for_each(sv, sv + linearRegressionMonomialSize, [&](DataType const s) { sum += s * (*rowdata++); });
295 
296  qFvalue[i] = sum;
297  }
298 
299  return true;
300 }
