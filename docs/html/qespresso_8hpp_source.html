359 bool qespresso::getSpeciesInformation()
360 {
361  if (!getTotalNumberRunFiles())
362  {
363  if (fullFileName.size())
364  {
365  UMUQFAILRETURN("There is no input file to open!");
366  }
367  UMUQFAILRETURN("Please set the file name!");
368  }
369  {
370  // Get an instance of the io object
371  umuq::io file;
372 
373  // File name
374  char fileName[LINESIZE];
375 
376  sprintf(fileName, fullFileName.c_str(), startIndexRunFiles);
377  if (file.openFile(fileName))
378  {
379  // Get an instance of a parser object to parse
380  umuq::parser p;
381  // counter for the number of species
382  nSpecies = 0;
383  // Counter for the number of species types
384  nSpeciesTypes = 0;
385 
386  // Read each line in the file and skip all the commented and empty line with the default comment "#"
387  while (file.readLine())
388  {
389  // Parse the line into line arguments
390  p.parse(file.getLine());
391  if (p.at<std::string>(0) == "set")
392  {
393  if (p.at<std::string>(1) == "NATOMS")
394  {
395  nSpecies = p.at<int>(3);
396  }
397  if (p.at<std::string>(1) == "NATOM")
398  {
399  if (p.at<std::string>(2) == "TYPES")
400  {
401  nSpeciesTypes = p.at<int>(4);
402  }
403  }
404  if (nSpecies && nSpeciesTypes)
405  {
406  break;
407  }
408  }
409  }
410  {
411  // Create an instance of the species object
412  umuq::species s;
413  // Read each line in the file and skip all the commented and empty line with the default comment "#"
414  while (file.readLine())
415  {
416  // Parse the line into line arguments
417  p.parse(file.getLine());
418  if (p.at<std::string>(0).substr(0, 2) == "Z:")
419  {
420  auto index = p.at<int>(1);
421  Species.push_back(s.getSpecies(index));
422  if (Species.size() == nSpeciesTypes)
423  {
424  break;
425  }
426  }
427  }
428  if (!Species.size())
429  {
430  file.rewindFile();
431  // Read each line in the file and skip all the commented and empty line with the default comment "#"
432  while (file.readLine())
433  {
434  // Parse the line into line arguments
435  p.parse(file.getLine());
436  if (p.at<std::string>(0) == "Number")
437  {
438  if (p.getLineArgNum() > 5)
439  {
440  if (p.at<std::string>(6).substr(0, 2) == "Z:")
441  {
442  auto index = p.at<int>(7);
443  Species.push_back(s.getSpecies(index));
444  if (Species.size() == nSpeciesTypes)
445  {
446  break;
447  }
448  }
449  }
450  }
451  }
452  }
453  }
454  file.closeFile();
455 
456  // Set the total number of steps to zero
457  totalNumberSteps = 0;
458 
459  // Loop through the files and count number of steps
460  for (std::size_t fileID = startIndexRunFiles; fileID < startIndexRunFiles + getTotalNumberRunFiles(); fileID++)
461  {
462  sprintf(fileName, fullFileName.c_str(), fileID);
463  if (file.isFileExist(fileName))
464  {
465  if (file.openFile(fileName))
466  {
467  std::size_t stepFlag = 0;
468  // Read each line in the file and skip all the commented and empty line with the default comment "#"
469  while (file.readLine())
470  {
471  if (stepFlag == 3)
472  {
473  totalNumberSteps++;
474  stepFlag = 0;
475  }
476  // Parse the line into line arguments
477  p.parse(file.getLine());
478  if (p.at<std::string>(0) == "v1")
479  {
480  stepFlag = 1;
481  continue;
482  }
483  if (p.at<std::string>(0) == "AtomId")
484  {
485  auto Id = 1;
486  while (file.readLine())
487  {
488  // Parse the line into line arguments
489  p.parse(file.getLine());
490  if (p.at<std::string>(0) == "AtomId")
491  {
492  Id++;
493  }
494  if (Id == nSpecies)
495  {
496  break;
497  }
498  }
499  stepFlag++;
500  continue;
501  }
502  }
503  file.closeFile();
504  }
505  continue;
506  }
507  return false;
508  }
509  if (nSpeciesTypes > 1)
510  {
511  if (file.isFileExist(parameterFileName))
512  {
513  if (file.openFile(parameterFileName))
514  {
515  // Read each line in the file and skip all the commented and empty line with the default comment "#"
516  while (file.readLine())
517  {
518  // Parse the line into line arguments
519  p.parse(file.getLine());
520  if (p.at<std::string>(0) == "set")
521  {
522  if (p.at<std::string>(1) == "ATOMIC")
523  {
524  auto str = file.getLine();
525  p.parse(str.replace(str.find_first_of("="), 1, " "));
526  sprintf(fileName, "%s%s", dataRootDirectory.c_str(), p.at<std::string>(4).c_str());
527  break;
528  }
529  }
530  }
531  file.closeFile();
532  if (file.isFileExist(fileName))
533  {
534  if (file.openFile(fileName))
535  {
536  // Create an instance of the species object
537  umuq::species s;
538 
539  // Read each line in the file and skip all the commented and empty line with the default comment "#"
540  while (file.readLine())
541  {
542  // Parse the line into line arguments
543  p.parse(file.getLine());
544  auto index = p.at<int>(0);
545  auto sp = s.getSpecies(index);
546  for (auto i = 0; i < nSpeciesTypes; i++)
547  {
548  if (sp.name == Species[i].name)
549  {
550  speciesTypes.push_back(i);
551  break;
552  }
553  }
554  }
555  file.closeFile();
556  return (speciesTypes.size() == nSpecies);
557  }
558  }
559  }
560  }
561  return false;
562  }
563  else
564  {
565  speciesTypes.resize(nSpecies, 0);
566  }
567  return true;
568  }
569  UMUQFAILRETURN("There is no input file to open!");
570  }

573 bool qespresso::dump(std::string const &baseCoordinatesFileName, std::string const &baseForcesFileName, std::string const &format)
574 {
575  if (!totalNumberSteps)
576  {
577  if (!getSpeciesInformation())
578  {
579  UMUQFAILRETURN("Failed to read the information from input files!")
580  }
581  }

583  // Get an instance of the io object
584  umuq::io file;
585 
586  // Get an instance of the io object
587  umuq::io outputFile;
588 
589  // Get an instance of the io object
590  umuq::io xyzFile;
591 
592  // Create flags to indicate the requested tasks
593  bool const coordsFileFlag = baseCoordinatesFileName.size();
594  bool const forceFileFlag = baseForcesFileName.size();
595  bool const xyzFileFlag = (format == "xyz") ? xyzFile.openFile("history.xyz", xyzFile.out) : false;
596 
597  // File name
598  char fileName[LINESIZE];
599 
600  // Get an instance of the units class with METAL style
601  umuq::units mdUnits("METAL");
602 
603  // Set the unit style for conversion from DFT-FE ELECTRON style to METAL
604  if (!mdUnits.convertFromStyle("ELECTRON"))
605  {
606  UMUQFAILRETURN("Can not set the conversion factors!");
607  }
608 
609  // Unit cell coordinates
610  std::vector<double> boundingVectors(9);
611 
612  // Species coordinates
613  std::vector<double> fractionalCoordinates(nSpecies * 3);
614 
615  // Species force
616  std::vector<double> force(nSpecies * 3);
617 
618  // Number of steps counter
619  std::size_t numberSteps = 0;
620 
621  // Loop through all the files
622  for (std::size_t fileID = startIndexRunFiles; fileID < startIndexRunFiles + getTotalNumberRunFiles(); fileID++)
623  {
624  sprintf(fileName, fullFileName.c_str(), fileID);
625  if (file.openFile(fileName))
626  {
627  // An instance of a parser object to parse
628  umuq::parser p;
629 
630  // Set an index indicator for coordinates and forces
631  std::size_t Id = 0;
632 
633  // Set the dumping flag
634  std::size_t dumpFlag = 0;
635 
636  while (file.readLine())
637  {
638  if (dumpFlag == 3)
639  {
640  if (coordsFileFlag)
641  {
642  std::string FileName = baseCoordinatesFileName + "%01d";
643  sprintf(fileName, FileName.c_str(), numberSteps);
644  if (outputFile.openFile(fileName, outputFile.out))
645  {
646  // Write the matrix in it
647  outputFile.saveMatrix<double>(boundingVectors, 3, 3);
648  outputFile.closeFile();
649  }
650 
651  FileName = "X" + baseCoordinatesFileName + "%01d";
652 
653  sprintf(fileName, FileName.c_str(), numberSteps);
654  if (outputFile.openFile(fileName, outputFile.out))
655  {
656  // Write the matrix in it
657  outputFile.saveMatrix<double>(fractionalCoordinates, nSpecies, 3);
658  outputFile.closeFile();
659  }
660  }
661 
662  if (forceFileFlag)
663  {
664  std::string FileName = baseForcesFileName + "%01d";
665  sprintf(fileName, FileName.c_str(), numberSteps);
666  if (outputFile.openFile(fileName, outputFile.out))
667  {
668  // Write the matrix in it
669  outputFile.saveMatrix<double>(force, nSpecies, 3);
670  outputFile.closeFile();
671  }
672  }
673 
674  if (xyzFileFlag)
675  {
676  // Get the IO file base stream
677  auto &fs = xyzFile.getFstream();
678  fs << nSpecies << "\n";
679  // fs << "step\t" << numberSteps << "\n";
680  fs << "Lattice=\"";
681  for (auto c : boundingVectors)
682  {
683  fs << " " << c;
684  }
685  fs << "\" Properties=species:S:1:pos:R:3\n";
686  auto coordsIt = fractionalCoordinates.begin();
687  for (auto speciesIndex = 0; speciesIndex < nSpecies; speciesIndex++)
688  {
689  fs << Species[speciesTypes[speciesIndex]].name << "\t";
690  fs << *coordsIt++ << " ";
691  fs << *coordsIt++ << " ";
692  fs << *coordsIt++ << "\n";
693  }
694  }
695 
696  numberSteps++;
697  dumpFlag = 0;
698  }
699 
700  // Parse the line into line arguments
701  p.parse(file.getLine());
702  if (p.at<std::string>(0) == "v1")
703  {
704  boundingVectors[0] = p.at<double>(1);
705  boundingVectors[1] = p.at<double>(2);
706  boundingVectors[2] = p.at<double>(3);
707  if (file.readLine())
708  {
709  // Parse the line into line arguments
710  p.parse(file.getLine());
711  boundingVectors[3] = p.at<double>(1);
712  boundingVectors[4] = p.at<double>(2);
713  boundingVectors[5] = p.at<double>(3);
714  if (file.readLine())
715  {
716  // Parse the line into line arguments
717  p.parse(file.getLine());
718  boundingVectors[6] = p.at<double>(1);
719  boundingVectors[7] = p.at<double>(2);
720  boundingVectors[8] = p.at<double>(3);
721  }
722  }
723 
724  // Convert from DFT-FE style to the output METAL style
725  mdUnits.convert<UnitType::Length>(boundingVectors);
726 
727  dumpFlag = 1;
728  Id = 0;
729  continue;
730  }
731 
732  if (p.at<std::string>(0) == "AtomId")
733  {
734  if (Id)
735  {
736  Id = 0;
737  force[Id++] = p.at<double>(2);
738  force[Id++] = p.at<double>(3);
739  force[Id++] = p.at<double>(4);
740  while (file.readLine())
741  {
742  // Parse the line into line arguments
743  p.parse(file.getLine());
744  if (p.at<std::string>(0) == "AtomId")
745  {
746  force[Id++] = p.at<double>(2);
747  force[Id++] = p.at<double>(3);
748  force[Id++] = p.at<double>(4);
749  continue;
750  }
751  if (Id / 3 == nSpecies)
752  {
753  break;
754  }
755  }
756 #ifdef DEBUG
757  if (Id / 3 != nSpecies)
758  {
759  UMUQFAILRETURN("There is a mismatch in the number of force data points = ", Id / 3, "!= ", nSpecies, " !");
760  }
761 #endif
762  // Convert from DFT-FE style to the output METAL style
763  mdUnits.convert<UnitType::Force>(force);
764  dumpFlag++;
765 
766  continue;
767  }
768  else
769  {
770  fractionalCoordinates[Id++] = p.at<double>(2);
771  fractionalCoordinates[Id++] = p.at<double>(3);
772  fractionalCoordinates[Id++] = p.at<double>(4);
773  while (file.readLine())
774  {
775  // Parse the line into line arguments
776  p.parse(file.getLine());
777  if (p.at<std::string>(0) == "AtomId")
778  {
779  fractionalCoordinates[Id++] = p.at<double>(2);
780  fractionalCoordinates[Id++] = p.at<double>(3);
781  fractionalCoordinates[Id++] = p.at<double>(4);
782  continue;
783  }
784  if (Id / 3 == nSpecies)
785  {
786  break;
787  }
788  }
789 #ifdef DEBUG
790  if (Id / 3 != nSpecies)
791  {
792  UMUQFAILRETURN("There is a mismatch in the number of coordinates data points = ", Id / 3, "!= ", nSpecies, " !");
793  }
794 #endif
795  // Convert from DFT-FE style to the output style
796  mdUnits.convert<UnitType::Length>(fractionalCoordinates);
797  dumpFlag++;
798 
799  {
800  // Creat an instance of a lattice object with the input bounding vectors
801  umuq::lattice l(boundingVectors);
802 
803  // Convert the fractional coordinates to the Cartesian coordinates
804  fractionalCoordinates = l.fractionalToCartesian(fractionalCoordinates);
805  }
806 
807  continue;
808  }
809  }
810  }
811 
812  file.closeFile();
813 
814  continue;
815  }
816  UMUQFAILRETURN("Failed to open the file!")
817  }
818  if (numberSteps != totalNumberSteps)
819  {
820  UMUQFAILRETURN("The read number of steps = ", numberSteps, "!= ", totalNumberSteps, " !");
821  }
822  if (xyzFileFlag)
823  {
824  xyzFile.closeFile();
825  }
826  return true;
827  }

830 bool qespresso::calculateMeanSquareDisplacement(std::size_t const speciesTypeId, double const timeStep)
831 {
832  if (!totalNumberSteps)
833  {
834  if (!getSpeciesInformation())
835  {
836  UMUQFAILRETURN("Failed to read the information from input files!")
837  }
838  }
839 
840  if (speciesTypeId > nSpeciesTypes)
841  {
842  UMUQFAILRETURN("The requested species index (", speciesTypeId, ") does not exist!");
843  }
844 
845  // Count the number of species of the requested type
846  auto nSpeciesOfRequestedType = 0;
847  std::for_each(speciesTypes.begin(), speciesTypes.end(), [&](int const s_i) { if (s_i == speciesTypeId) nSpeciesOfRequestedType++; });
848 
849  if (!nSpeciesOfRequestedType)
850  {
851  UMUQFAILRETURN("The is no species of the requested type in the data!");
852  }
853 
854  {
855  // Get an instance of the io object
856  umuq::io file;
857 
858  // File name
859  char fileName[LINESIZE];
860 
861  // Initialize msd arrays for computing msd
862  std::vector<double> msd(nSpeciesOfRequestedType * totalNumberSteps, 0.0);
863  std::vector<double> rr2(totalNumberSteps, 0.0);
864  std::vector<int> msm(totalNumberSteps, 0);
865 
866  {
867  // Get an instance of the units class with METAL style
868  umuq::units mdUnits("METAL");
869 
870  // Set the unit style for conversion from DFT-FE ELECTRON style to METAL
871  if (!mdUnits.convertFromStyle("ELECTRON"))
872  {
873  UMUQFAILRETURN("Can not set the conversion factors!");
874  }
875 
876  // Creat an instance of a lattice object with the input bounding vectors
877  umuq::lattice dftLattice;
878 
879  // Number of steps counter
880  std::size_t numberSteps = 0;
881 
882  // Bounding vectors
883  std::vector<double> boundingVectors(9);
884 
885  // Temporary arrays
886  std::vector<double> inCoordinates(nSpeciesOfRequestedType * 3);
