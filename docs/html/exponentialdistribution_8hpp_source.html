215 {
216  for (std::size_t i = 0; i < this->numParams; i++)
217  {
218  if (x[i] < RealType{})
219  {
220  return RealType{};
221  }
222  }
223  RealType sum(1);
224  for (std::size_t i = 0; i < this->numParams; i++)
225  {
226  sum *= std::exp(-x[i] / this->params[i]) / this->params[i];
