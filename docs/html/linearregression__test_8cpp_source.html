175 bool meshgrid(T *&inDataPt, int const nPoints, int const nDim, double *Lb, double *Ub)
176 {
177  if (nDim < 1)
178  {
179  std::cerr << "Error : " << __FILE__ << ":" << __LINE__ << " : " << std::endl;
180  std::cerr << " Wrong dimension ! " << std::endl;
181  return false;
182  }
183 
184  if (inDataPt == nullptr)
185  {
186  try
187  {
188  inDataPt = new T[nPoints * nDim];
189  }
190  catch (std::bad_alloc &e)
191  {
192  std::cerr << "Error : " << __FILE__ << ":" << __LINE__ << " : " << std::endl;
193  std::cerr << " Failed to allocate memory : " << e.what() << std::endl;
194  return false;
195  }
196  }
197 
198  // std::random_device rd;
199  std::mt19937 gen(123);
200  std::uniform_real_distribution<> dis(0.0, 1.0);
201 
202  switch (nDim)
203  {
204  case 1:
205  for (int i = 0; i < nPoints; i++)
206  {
207  inDataPt[i] = Lb[0] + dis(gen) * (Ub[0] - Lb[0]);
208  }
209  break;
210 
211  case 2:
212  for (int i = 0, n = 0; i < nPoints; i++)
213  {
214  inDataPt[n] = Lb[0] + dis(gen) * (Ub[0] - Lb[0]);
215  inDataPt[n + 1] = Lb[1] + dis(gen) * (Ub[1] - Lb[1]);
216  n += 2;
217  }
218  break;
219 
220  default:
221  for (int i = 0, n = 0; i < nPoints; i++)
222  {
223  for (int d = 0; d < nDim; d++, n++)
224  {
225  inDataPt[n] = Lb[d] + dis(gen) * (Ub[d] - Lb[d]);
226  }
227  }
228  break;
229  }
230 
231  return true;
232 }
