203 template <typename DataType>
204 bool simplexNM<DataType>::iterate()
205 {
206  // Simplex iteration tries to minimize function f value
207  // ws1 and ws2 vectors store tried corner point coordinates
208 
209  int hi(0);
210  int s_hi;
211  int lo(0);
212 
213  DataType dhi;
214  DataType dlo;
215  DataType ds_hi;
216 
217  DataType val;
218  DataType val2;
219 
220  // Get index of highest, second highest and lowest point
221  dhi = dlo = y1[0];
222  ds_hi = y1[1];
223  s_hi = 1;
224 
225  int const n = this->getDimension();
226 
227  for (int i = 1; i < n + 1; i++)
228  {
229  val = y1[i];
230  if (val < dlo)
231  {
232  dlo = val;
233  lo = i;
234  }
235  else if (val > dhi)
236  {
237  ds_hi = dhi;
238  s_hi = hi;
239  dhi = val;
240  hi = i;
241  }
242  else if (val > ds_hi)
243  {
244  ds_hi = val;
245  s_hi = i;
246  }
247  }
248 
249  // Reflect the highest value
250  val = moveCorner(-static_cast<DataType>(1), hi, this->ws1);
251 
252  if (std::isfinite(val) && val < y1[lo])
253  {
254 
255  // Reflected point becomes lowest point, try expansion
256  val2 = moveCorner(-static_cast<DataType>(2), hi, this->ws2);
257 
258  if (std::isfinite(val2) && val2 < y1[lo])
259  {
260  // Copy the elements of the vector ws2 into the hi-th row of the matrix x1
261  std::ptrdiff_t const Id = hi * n;
262 
263  std::copy(this->ws2.begin(), this->ws2.end(), x1.data() + Id);
264 
265  y1[hi] = val2;
266  }
267  else
268  {
269  // Copy the elements of the vector ws1 into the hi-th row of the matrix x1
270  std::ptrdiff_t const Id = hi * n;
271 
272  std::copy(this->ws1.begin(), this->ws1.end(), x1.data() + Id);
273 
274  y1[hi] = val;
275  }
276  }
277  // Reflection does not improve things enough
278  // or
279  // we got a non-finite (illegal) function value
280  else if (!std::isfinite(val) || val > y1[s_hi])
281  {
282  if (std::isfinite(val) && val <= y1[hi])
283  {
284 
285  // If trial point is better than highest point, replace highest point
286  // Copy the elements of the vector ws1 into the hi-th row of the matrix x1
287  std::ptrdiff_t const Id = hi * n;
288 
289  std::copy(this->ws1.begin(), this->ws1.end(), x1.data() + Id);
290 
291  y1[hi] = val;
292  }
293 
294  // Try one dimensional contraction
295  val2 = moveCorner(static_cast<DataType>(0.5), hi, this->ws2);
296 
297  if (std::isfinite(val2) && val2 <= y1[hi])
298  {
299  // Copy the elements of the vector ws2 into the hi-th row of the matrix x1
300  std::ptrdiff_t const Id = hi * n;
301 
302  std::copy(this->ws2.begin(), this->ws2.end(), x1.data() + Id);
303 
304  y1[hi] = val2;
305  }
306  else
307  {
308  // Contract the whole simplex in respect to the best point
309  if (!contractByBest(lo, this->ws1))
310  {
311  UMUQFAILRETURN("contractByBest failed!");
312  }
313  }
314  }
315  else
316  {
317  // Trial point is better than second highest point. Replace highest point by it
318  // Copy the elements of the vector ws1 into the hi-th row of the matrix x1
319  std::ptrdiff_t const Id = hi * n;
320 
321  std::copy(this->ws1.begin(), this->ws1.end(), x1.data() + Id);
322 
323  y1[hi] = val;
324  }
325 
326  // Return lowest point of simplex as x
327  lo = static_cast<int>(std::distance(y1.begin(), std::min_element(y1.begin(), y1.end())));
328 
329  // Copy the elements of the lo-th row of the matrix x1 into the vector x
330  {
331  std::ptrdiff_t const Id = lo * n;
332 
333  std::copy(x1.data() + Id, x1.data() + Id + n, this->x.begin());
334  }
335 
336  this->fval = y1[lo];
337 
338  // Update simplex size
339  this->characteristicSize = computeSize();
340 
341  return true;
342 }

345 DataType simplexNM<DataType>::moveCorner(DataType const coeff, int const corner, std::vector<DataType> &X)
346 {
347  int const n = this->getDimension();
348 
349  // We have a row-major order matrix \f$ (n+1) \times n \f$
350  for (int j = 0; j < n; j++)
351  {
352  DataType s(0);
353  for (int i = 0; i < n + 1; i++)
354  {
355  if (i != corner)
356  {
357  std::ptrdiff_t const Id = i * n + j;
358 
359  s += x1[Id];
360  }
361  }
362 
363  s /= static_cast<DataType>(n);
364 
365  std::ptrdiff_t const Idx = corner * n + j;
366 
367  DataType newval = s - coeff * (s - x1[Idx]);
368 
369  X[j] = newval;
370  }
371 
372  return this->fun.f(X.data());
373 }

375 template <typename DataType>
376 bool simplexNM<DataType>::contractByBest(int const best, std::vector<DataType> &X)
377 {
378  std::ptrdiff_t const n = static_cast<std::ptrdiff_t>(this->getDimension());
379  std::ptrdiff_t const b = static_cast<std::ptrdiff_t>(best);
380 
381  for (std::ptrdiff_t i = 0; i < n + 1; i++)
382  {
383  if (i != b)
384  {
385  DataType newval;
386 
387  std::ptrdiff_t Id = i * n;
388  std::ptrdiff_t Idb = b * n;
389 
390  for (std::ptrdiff_t j = 0; j < n; j++, Id++, Idb++)
391  {
392  newval = static_cast<DataType>(0.5) * (x1[Id] + x1[Idb]);
393  x1[Id] = newval;
394  }
395 
396  // Evaluate function in the new point
397  Id = i * n;
398 
399  // Copy the elements of the i-th row of the matrix x1 into the vector X
400  std::copy(x1.data() + Id, x1.data() + Id + n, X.begin());
401 
402  newval = this->fun.f(X.data());
403 
404  y1[i] = newval;
405 
406  // Notify caller that we found at least one bad function value.
407  // we finish the contraction (and do not abort) to allow the user
408  // to handle the situation
409  if (!std::isfinite(newval))
410  {
411  UMUQFAILRETURN("The iteration encountered a singular point where the function or its derivative evaluated to Inf or NaN!");
412  }
413  }
414  }
415 
416  return true;

442 template <typename DataType>
443 DataType simplexNM<DataType>::computeSize()
444 {
445  // Calculates simplex size as average sum of length of vectors
446  // from simplex center to corner points:
447  // \f$ (sum(|| y - y_middlepoint ||)) / n * \f$
448 
449  int const n = this->getDimension();
450 
451  // Calculate middle point
452  computeCenter(this->ws2);
453 
454  DataType s(0);
455  for (int i = 0; i < n + 1; i++)
456  {
457  std::ptrdiff_t const Id = i * n;
458 
459  // Copy the elements of the i-th row of the matrix x1 into the vector ws1
460  std::copy(x1.data() + Id, x1.data() + Id + n, this->ws1.begin());
461 
462  for (int j = 0; j < n; j++)
463  {
464  this->ws1[j] -= this->ws2[j];
465  }
466 
467  {
468  //Compute the Euclidean norm \f$ ||x||_2 = \sqrt {\sum x_i^2} of the vector x = gradient. \f$
469  DataType sum(0);
470  std::for_each(this->ws1.begin(), this->ws1.end(), [&](DataType const w_i) { sum += w_i * w_i; });
471 
472  s += std::sqrt(sum);
473  }
474  }
475 
476  return s / static_cast<DataType>(n + 1);
477 }
