357 bool utility::tryLockOrSleep(int const millisecondsDuration, int const maxNumTrials)
358 {
359  int numTrials(0);
360  while (numTrials < maxNumTrials)
361  {
362  // Try to lock mutex to modify 'job_shared'
363  if (utility_m.try_lock())
364  {
365  return true;
366  }
367 
368  // Stops the execution of the current thread for a specified time duration
369  std::this_thread::sleep_for(std::chrono::milliseconds(millisecondsDuration));
370  numTrials++;
371  }
372 #ifdef DEBUG
373  UMUQWARNING("Failed to lock the mutex after [", numTrials, "] trials!");
374 #endif
375  return false;
376 }
