638 bool radialBasisFunction<kernelType, polynomialTailType>::addPoint(EMatrixXd const &Points, EVectorXd const &FunctionValues)
639 {
640  // check for the correct number of points and function values at those points
641  UMUQ_assert(Points.cols() == FunctionValues.rows());
642 
643  if (!nPoints)
644  {
645  return setPoints(Points, FunctionValues);
646  }
647 
648  int const newNumPoints = static_cast<int>(FunctionValues.rows());
649  if (newNumPoints < 2)
650  {
651  if (newNumPoints)
652  {
653  return addPoint(Points.col(0), FunctionValues(0));
654  }
655  UMUQFAILRETURN("There is no input data!");
656  }
657 
658  if (nPoints + newNumPoints > maxNumPoints)
659  {
660  UMUQFAILRETURN("Capacity exceeded ! \n Number of current points ", nPoints, " + number of new points ", newNumPoints, " > the available capacity ", maxNumPoints);
661  }
662 
663 #ifdef DEBUG
664  gTimer.tic();
665 #endif
666 
667  // Map point to be in the unit box
668  EMatrixXd points = scaleToUnitBox(Points, lowerBounds, upperBounds);
669 
670  int const n = polynomialSpaceDimension + nPoints;
671 
672  EMatrixXd px = tail.evaluate(points);
673  EMatrixXd B = EMatrixXd::Zero(n, newNumPoints);
674 
675  B.block(0, 0, polynomialSpaceDimension, newNumPoints) = px;
676  B.block(polynomialSpaceDimension, 0, nPoints, newNumPoints) = kernel.evaluate(L2Distance<double, true>(interpolationNodes.block(0, 0, nDimensions, nPoints), points));
677 
678  EMatrixXd K = kernel.evaluate(L2Distance(points));
679 
680  // REGULARIZATION
681  K += dampingCoefficient * EMatrixXd::Identity(newNumPoints, newNumPoints);
682 
683  // Update the LU factorization
684  // Step 1
685  EMatrixXd U12 = Eigen::PermutationWrapper<EVectorX<int>>(permutationP.head(n)) * B;
686  EMatrixXd L21 = U12;
687 
688  // Step 2
689  matrixLU.block(0, 0, n, n).template triangularView<Eigen::UnitLower>().solveInPlace(U12);
690 
691  // Step 3
692  matrixLU.block(0, 0, n, n).template triangularView<Eigen::Upper>().solveInPlace(L21);
693 
694  EMatrixXd C;
695 
696  // Standard Cholesky decomposition (LL^T)
697  Eigen::LLT<EMatrixXd> llt(K - L21.transpose() * U12);
698 
699  if (llt.info() == Eigen::Success)
700  {
701  // retrieve factor L in the decomposition
702  C = llt.matrixL();
703  }
704  else
705  {
706  UMUQWARNING("Cholesky factorization failed, computing new LU from scratch...");
707 
708  // Add new points
709  functionValues.segment(n, newNumPoints) = FunctionValues;
710  interpolationNodes.block(0, nPoints, nDimensions, newNumPoints) = points;
711  nPoints += newNumPoints;
712  // Build LU from scratch
713  return setPoints(getCurrentPoints(), functionValues.segment(polynomialSpaceDimension, nPoints));
714  }
715 
716  matrixLU.block(n, 0, newNumPoints, n).template triangularView<Eigen::UnitLower>() = L21;
717  matrixLU.block(n, n, newNumPoints, newNumPoints).template triangularView<Eigen::UnitLower>() = C;
718  matrixLU.block(0, n, n, newNumPoints).template triangularView<Eigen::Upper>() = U12;
719  matrixLU.block(n, n, newNumPoints, newNumPoints).template triangularView<Eigen::Upper>() = C.transpose();
720 
721  permutationP.segment(n, newNumPoints) = EVectorX<int>::LinSpaced(newNumPoints, n, n + newNumPoints - 1);
722 
723  // Update F and add the centers
724  functionValues.segment(n, newNumPoints) = FunctionValues;
725 
726  interpolationNodes.block(0, nPoints, nDimensions, newNumPoints) = points;
727 
728  nPoints += newNumPoints;
729 
730  shouldRecomputeCoefficients = true;
731 
732 #ifdef DEBUG
733  gTimer.toc("radialBasisFunction->addPoint");
734 #endif
735 
736  return true;
737 }

811 template <class kernelType, class polynomialTailType>
812 EVectorXd radialBasisFunction<kernelType, polynomialTailType>::deriv(EVectorXd const &Point) const
813 {
814  if (shouldRecomputeCoefficients)
815  {
816  UMUQFAIL("radialBasisFunction is not updated. You need to call fit() first");
817  }
818  // Map point to be in the unit box
819  EVectorXd point = scaleToUnitBox(Point, lowerBounds, upperBounds);
820 
821  EMatrixXd dpx = tail.deriv(point);
822  EVectorXd Distances = L2Distance<double>(point, interpolationNodes.block(0, 0, nDimensions, nPoints));
823  // Better safe than sorry
824  for (int i = 0; i < nPoints; ++i)
825  {
826  if (Distances(i) < 1e-10)
827  {
828  Distances(i) = 1e-10;
829  }
830  }
831  EMatrixXd dsx = -interpolationNodes.block(0, 0, nDimensions, nPoints);
832  dsx.colwise() += point;
833  dsx.array().rowwise() *= coefficientVector.segment(polynomialSpaceDimension, nPoints).cwiseProduct(kernel.deriv(Distances)).cwiseProduct(Distances.cwiseInverse()).array().transpose();
834  return dsx.rowwise().sum() + dpx.transpose() * coefficientVector.head(polynomialSpaceDimension);

837 template <class kernelType, class polynomialTailType>
838 bool radialBasisFunction<kernelType, polynomialTailType>::fit()
839 {
840  if (nPoints < polynomialSpaceDimension)
841  {
842  UMUQFAILRETURN("Current number of points = ", nPoints, " < ", polynomialSpaceDimension, " (polynomial dimension), Not enough points!");
843  }
844 #ifdef DEBUG
845  gTimer.tic();
846 #endif
847  if (shouldRecomputeCoefficients)
848  {
849  shouldRecomputeCoefficients = false;
850 
851  int const n = nPoints + polynomialSpaceDimension;
852 
853  /* The decomposition PA = LU.
854  * So we proceed as follows:
855  * Step 1: compute c = Pb.
856  * Step 2: replace c by the solution x to Lx = c.
857  * Step 3: replace c by the solution x to Ux = c.
858  */
859 
860  // Step 1
861  coefficientVector.head(n) = Eigen::PermutationWrapper<EVectorX<int>>(permutationP.head(n)) * functionValues.head(n);
862 
863  // Step 2
864  matrixLU.block(0, 0, n, n).template triangularView<Eigen::UnitLower>().solveInPlace(coefficientVector.head(n));
865 
866  // Step 3
867  matrixLU.block(0, 0, n, n).template triangularView<Eigen::Upper>().solveInPlace(coefficientVector.head(n));
868  }
869 #ifdef DEBUG
870  gTimer.toc("radialBasisFunction->fit");
871 #endif

1020 template <class kernelType, class polynomialTailType>
1021 bool radialBasisFunctionCap<kernelType, polynomialTailType>::fit()
1022 {
1023  if (this->nPoints < this->polynomialSpaceDimension)
1024  {
1025  UMUQFAILRETURN("Current number of points = ", this->nPoints, " < ", this->polynomialSpaceDimension, " (polynomial dimension), Not enough points!");
1026  }
1027 #ifdef DEBUG
1028  gTimer.tic();
1029 #endif
1030  if (this->shouldRecomputeCoefficients)
1031  {
1032  double functionValuesMedian;
1033  // Computes the median
1034  {
1035  // We do partial sorting algorithm that rearranges elements
1036  std::vector<double> functionValuesVector(this->nPoints);
1037  // Copy the function values to the std vector
1038  EMap<EVectorXd>(functionValuesVector.data(), this->functionValues.segment(this->polynomialSpaceDimension, this->nPoints));
1039  std::nth_element(functionValuesVector.begin(), functionValuesVector.begin() + this->nPoints / 2, functionValuesVector.end());
1040  functionValuesMedian = functionValuesVector[this->nPoints / 2];
1041  std::cout << functionValuesMedian << std::endl;
1042  }
1043 
1044  int const n = this->nPoints + this->polynomialSpaceDimension;
1045 
1046  for (int i = this->polynomialSpaceDimension; i < n; ++i)
1047  {
1048  // Apply the capping
1049  if (this->functionValues(i) > functionValuesMedian)
1050  {
1051  this->functionValues(i) = functionValuesMedian;
1052  }
1053  }
1054 
1055  /* The decomposition PA = LU.
1056  * So we proceed as follows:
1057  * Step 1: compute c = Pb.
1058  * Step 2: replace c by the solution x to Lx = c.
1059  * Step 3: replace c by the solution x to Ux = c.
1060  */
1061 
1062  // Step 1
1063  this->coefficientVector.head(n) = Eigen::PermutationWrapper<EVectorX<int>>(this->permutationP.head(n)) * this->functionValues.head(n);
1064 
1065  // Step 2
1066  this->matrixLU.block(0, 0, n, n).template triangularView<Eigen::UnitLower>().solveInPlace(this->coefficientVector.head(n));
1067 
1068  // Step 3
1069  this->matrixLU.block(0, 0, n, n).template triangularView<Eigen::Upper>().solveInPlace(this->coefficientVector.head(n));
1070 
1071  this->shouldRecomputeCoefficients = false;
1072  }
1073 #ifdef DEBUG
1074  gTimer.toc("radialBasisFunctionCap->fit");
1075 #endif
1076  return true;
1077 }
