292 lattice::lattice(std::vector<double> const &boundingVectors) : latticeType(LatticeType::NONE)
293 {
294  if (boundingVectors.size() != 9)
295  {
296  UMUQFAIL("The bounding vectors with wrong size ", boundingVectors.size(), " != 9 !");
297  }
298 
299  EVectorXd BoundingVectors = EVectorMapTypeConst<double>(boundingVectors.data(), 9);
300 
301  // Compute the length of the bounding vectors
302  basisVectorLength[0] = BoundingVectors.segment<3>(0).norm();
303  basisVectorLength[1] = BoundingVectors.segment<3>(3).norm();
304  basisVectorLength[2] = BoundingVectors.segment<3>(6).norm();
305  if ((basisVectorLength.array() < UTHRESHOLD).any())
306  {
307  UMUQFAIL("The input vector length should be positive number bigger than UMUQ threshold!");
308  }
309 
310  // Normalize the bounding vectors
311  BoundingVectors.segment<3>(0) /= basisVectorLength[0];
312  BoundingVectors.segment<3>(3) /= basisVectorLength[1];
313  BoundingVectors.segment<3>(6) /= basisVectorLength[2];
314 
315  // Compute the angles between bounding vectors
316  double dot = BoundingVectors.segment<3>(3).dot(BoundingVectors.segment<3>(6));
317  dot = (dot < -1.0 ? -1.0 : (dot > 1.0 ? 1.0 : dot));
318  basisVectorAngle[0] = std::acos(dot);
319 
320  dot = BoundingVectors.segment<3>(0).dot(BoundingVectors.segment<3>(6));
321  dot = (dot < -1.0 ? -1.0 : (dot > 1.0 ? 1.0 : dot));
322  basisVectorAngle[1] = std::acos(dot);
323 
324  dot = BoundingVectors.segment<3>(0).dot(BoundingVectors.segment<3>(3));
325  dot = (dot < -1.0 ? -1.0 : (dot > 1.0 ? 1.0 : dot));
326  basisVectorAngle[2] = std::acos(dot);
327 
328  realSpaceBasisVector = EVectorXd::Zero(9);
329  reciprocalSpaceBasisVector = EVectorXd::Zero(9);
330 
331  init();
332 }

400 void lattice::init()
401 {
402  // Compute the Cos and Sin of the basis vector angles
403  EVector3d basisVectorAngleCos = basisVectorAngle.array().cos();
404  EVector3d basisVectorAngleSin = basisVectorAngle.array().sin();
405 
406  // For the first axis project the length along the x-axis
407  realSpaceBasisVector[0] = basisVectorLength[0];
408 
409  // For the second axis keep it in the xy-plane and rotate through angle between axes
410  realSpaceBasisVector[3] = basisVectorLength[1] * basisVectorAngleCos[2];
411  realSpaceBasisVector[4] = basisVectorLength[1] * basisVectorAngleSin[2];
412 
413  realSpaceBasisVector.segment<3>(3) = (UTHRESHOLD < realSpaceBasisVector.segment<3>(3).array().abs()).select(realSpaceBasisVector.segment<3>(3), 0.0);
414  // The third axis needs to rotate relative to the two previously-defined vectors.
415 
416  // (1) Rotate away from the x-axis by angle basisVectorAngle[1]
417  realSpaceBasisVector[6] = basisVectorLength[2] * basisVectorAngleCos[1];
418  realSpaceBasisVector[8] = basisVectorLength[2] * basisVectorAngleSin[1];
419 
420  // (2) Do a constrained rotation around the x-axis such that the angle becomes basisVectorAngle[0]
421  double const c = (basisVectorAngleCos[2] * basisVectorAngleCos[1] - basisVectorAngleCos[0]) / (basisVectorAngleSin[1] * basisVectorAngleSin[2]);
422 
423  realSpaceBasisVector[7] = -realSpaceBasisVector[8] * c;
424  realSpaceBasisVector[8] *= std::sqrt(1 - c * c);
425 
426  realSpaceBasisVector.segment<3>(6) = (UTHRESHOLD < realSpaceBasisVector.segment<3>(6).array().abs()).select(realSpaceBasisVector.segment<3>(6), 0.0);
427 
428  // Construct the reciprocal space basis
429  reciprocalSpaceBasisVector << realSpaceBasisVector.segment<3>(3).cross(realSpaceBasisVector.segment<3>(6)),
430  realSpaceBasisVector.segment<3>(6).cross(realSpaceBasisVector.segment<3>(0)),
431  realSpaceBasisVector.segment<3>(0).cross(realSpaceBasisVector.segment<3>(3));
432 
433  // Compute the Volume
434  volume = realSpaceBasisVector.segment<3>(0).dot(reciprocalSpaceBasisVector.segment<3>(0));
435 
436  double const reciprocalVolume = 1.0 / volume;
437 
438  reciprocalSpaceBasisVector *= reciprocalVolume;
439 
440  reciprocalSpaceBasisVector = (UTHRESHOLD < reciprocalSpaceBasisVector.array().abs()).select(reciprocalSpaceBasisVector, 0.0);
441 
442  // Create our metric:
443  std::size_t k = 0;
444  for (auto i = 0; i < 3; i++)
445  {
446  for (auto j = 0; j <= i; j++)
447  {
448  metricTensor[k++] = realSpaceBasisVector.segment<3>(i * 3).dot(realSpaceBasisVector.segment<3>(j * 3));
449  }
450  }
451 }

520

524  UMUQFAIL("Vector of points has a wrong dimension mod(", n, " , 3) != 0 !");
525  }
526 #endif
527  std::vector<double> Points(n);
528 
529  // If only we have one points
530  if (n == 3)
531  {
532  Points[0] = points[0] * reciprocalSpaceBasisVector[0] + points[1] * reciprocalSpaceBasisVector[1] + points[2] * reciprocalSpaceBasisVector[2];
533  Points[1] = points[0] * reciprocalSpaceBasisVector[3] + points[1] * reciprocalSpaceBasisVector[4] + points[2] * reciprocalSpaceBasisVector[5];
534  Points[2] = points[0] * reciprocalSpaceBasisVector[6] + points[1] * reciprocalSpaceBasisVector[7] + points[2] * reciprocalSpaceBasisVector[8];
535  }
536  else

539  EMapType<double> PointMatrix(Points.data(), n / 3, 3);
540  EMapTypeConst<double> reciprocalSpaceBasisMatrix(reciprocalSpaceBasisVector.data(), 3, 3);
541  PointMatrix = pointMatrix * reciprocalSpaceBasisMatrix;
542  }
543  for (auto i : Points)

548  }

600 double lattice::cartesianDistanceBetweenFractionalPoints(EVector3d const &point1, EVector3d const &point2)
601 {
602  EVector3d pointDiff = point1 - point2;
603  double dist = (pointDiff[0] * metricTensor[0] + pointDiff[1] * metricTensor[1] + pointDiff[2] * metricTensor[3]) * pointDiff[0] +
604  (pointDiff[0] * metricTensor[1] + pointDiff[1] * metricTensor[2] + pointDiff[2] * metricTensor[4]) * pointDiff[1] +
605  (pointDiff[0] * metricTensor[3] + pointDiff[1] * metricTensor[4] + pointDiff[2] * metricTensor[5]) * pointDiff[2];
606  return std::sqrt(dist);
607 }
