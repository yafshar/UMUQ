631 inline void Saru::advance(unsigned int steps)
632 {
633  // Computes the LCG advancement AND the Weyl D*E mod m simultaneously
634  unsigned int currentA = LCGA;
635  unsigned int currentC = LCGC;
636  unsigned int currentDelta = oWeylDelta;
637  unsigned int netDelta = 0;
638 
639  while (steps)
640  {
641  if (steps & 1)
642  {
643  state = currentA * state + currentC; // LCG step
644  netDelta += (netDelta < oWeylPeriod - currentDelta) ? currentDelta : currentDelta - oWeylPeriod;
645  }
646  // Change the LCG to step at twice the rate as before
647  currentC += currentA * currentC;
648  currentA *= currentA;
649  // Change the Weyl delta to step at 2X rate
650  currentDelta += (currentDelta < oWeylPeriod - currentDelta) ? currentDelta : currentDelta - oWeylPeriod;
651  steps /= 2;
652  }
653  // Apply the net delta to the Weyl state.
654  wstate += (wstate - oWeylOffset < oWeylPeriod - netDelta) ? netDelta : netDelta - oWeylPeriod;
655 }

658 inline Saru Saru::fork() const
659 {
660  /* These lines are FREE since they'll be precomputed at compile time. They
661  take user values like 1 2 3 and hash them to become roughly uncorrelated. */
662  static const unsigned int churned1 = 0xDEADBEEF ^ (0x1fc4ce47 * (seed1 ^ (seed1 >> 13)));
663  static const unsigned int churned2 = 0x1234567 + (0x82948463 * (churned1 ^ (churned1 >> 20)));
664  static const unsigned int churned3 = 0x87654321 ^ (0x87655677 * (churned2 ^ (churned2 >> 16)));
665  Saru z;
666  z.state = churned2 + state + (churned3 ^ wstate);
667  unsigned int add = (z.state + churned1) >> 1;
668  z.wstate += (z.wstate - oWeylOffset < oWeylPeriod - add) ? add : add - oWeylPeriod;
669  return z;
670 }
