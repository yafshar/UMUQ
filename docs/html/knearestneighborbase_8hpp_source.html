376 template <typename DataType, class FlannDistanceType>
377 void kNearestNeighborBase<DataType, FlannDistanceType>::buildIndex(DataType *idata)
378 {
379  flann::Matrix<DataType> dataset(idata, nDataPoints, dataDimension);
380 
381  // Construct an randomized kd-tree index using 4 kd-trees
382  // For the number of parallel kd-trees to use (Good values are in the range [1..16])
383  flann::Index<FlannDistanceType> index(dataset, flann::KDTreeIndexParams(4));
384  index.buildIndex();
385 
386  // Do a knn search, using 128 checks
387  // Number of checks means: How many leafs to visit when searching
388  // for neighbors (-1 for unlimited)
389  index.knnSearch(dataset, indices, dists, nNearestNeighborsToFind, flann::SearchParams(128));
390 }

470 template <typename DataType, class FlannDistanceType>
471 bool kNearestNeighborBase<DataType, FlannDistanceType>::checkNearestNeighbors()
472 {
473  if (the_same)
474  {
475  return true;
476  }
477  DataType const eps = std::numeric_limits<DataType>::epsilon();
478  std::size_t s(0);
479  for (std::size_t i = 0; i < nQueryDataPoints; ++i)
480  {
481  std::ptrdiff_t const Id = i * nNearestNeighborsToFind;
482  s += (dists_ptr[Id] < eps);
483  }
484  return (s != nQueryDataPoints);
485 }
