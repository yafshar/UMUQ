278 bool tmcmcStats::findOptimumP(double const *FunValues, int const nFunValues, double const PJ, double *OptimumP, double *OptimumCoefVar)
279 {
280  // Set the global helper pointer & variables
281  functionValues = const_cast<double *>(FunValues);
282  nFunctionValues = nFunValues;
283  pj = PJ;
284 
285  // Second, we have to set the function, input vector and stepsize in the function minimizer
286  if (!fMinimizer->set(CoefVarFun, &pj, &optParams.Step))
287  {
288  UMUQFAILRETURN("Failed to set the minimizer!");
289  }
290 
291  // Third, initialize the minimizer
292  if (!fMinimizer->init())
293  {
294  UMUQFAILRETURN("Failed to initialize the minimizer!");
295  }
296 
297  // Print the initial starting points, if it is requested
298  if (optParams.Display)
299  {
300  double *x = fMinimizer->getX();
301  UMUQMSG("x =", x[0]);
302  }
303 
304  // Forth, iterate until we reach the absolute tolerance
305 
306  // Fail:-1, Success:0, Continue:1
307  int status = 1;
308 
309  int iter(0);
310  while (iter < optParams.MaxIter && status == 1)
311  {
312  iter++;
313 
314  if (!fMinimizer->iterate())
315  {
316  UMUQFAILRETURN("Failed to iterate the minimizer!");
317  }
318 
319  if (optParams.Display)
320  {
321  double *x = fMinimizer->getX();
322  UMUQMSG(iter, ": ");
323  UMUQMSG("CoefVar(p=", x[0], ") =", fMinimizer->getMin(), ", & characteristic size =", fMinimizer->size());
324  }
325 
326  status = fMinimizer->testSize(optParams.Tolerance);
327  }
328 
329  // If it did not fail
330  if (status == 0 || status == 1)
331  {
332  OptimumP = fMinimizer->getX();
333  *OptimumCoefVar = fMinimizer->getMin();
334 
335  if (optParams.Display)
336  {
337  UMUQMSG(fMinimizer->getName(), ", on CoefVarFun : ", iter, " iters, CoefVar(p)=", *OptimumCoefVar);
338  UMUQMSG(((status == 0) ? "Converged to minimum at p = " : "Stopped at p = "));
339  UMUQMSG(*OptimumP);
340  }
341  return (status ? (std::abs(fMinimizer->getMin()) <= optParams.Tolerance) : true);
342  }
343  UMUQFAILRETURN("Could not find the optimum value of p through optimization!");
344 }

346 bool tmcmcStats::searchOptimumP(double const *FunValues, int const nFunValues, double const PJ, double *OptimumP, double *OptimumCoefVar)
347 {
348  double MinValp = double{};
349  double MaxValp = double{4};
350 
351  // Initialize the step size to the prescribed user value
352  double stepSize(optParams.Step);
353 
354  for (;;)
355  {
356  if (optParams.Display)
357  {
358  UMUQMSG("Search for optimum p in [", MinValp, ", ", MaxValp, "] with step size =", stepSize);
359  }
360 
361  // Define some variables to use in the search
362  bool found(false);
363 
364  // Maximum number of iterations
365  std::size_t const MaxIter = static_cast<std::size_t>((MaxValp - MinValp) / stepSize);
366 
367  // Optimum value of p
368  double optimumP(MinValp);
369 
370  // Max value of CoefVar
371  double MaxCoefVar(std::numeric_limits<double>::max());
372 
373  // We search the space for the optimum value
374  for (std::size_t iter = 0; iter < MaxIter; iter++)
375  {
376  double const p = MinValp + iter * stepSize;
377  double const NewCoefVar = CoefVarFun(&p);
378 
379  if (NewCoefVar < MaxCoefVar)
380  {
381  MaxCoefVar = NewCoefVar;
382  optimumP = p;
383 
384  if (MaxCoefVar <= coefVarPresetThreshold)
385  {
386  found = true;
387  break;
388  }
389  }
390  }
391 
392  // Success
393  if (found)
394  {
395  *OptimumP = optimumP;
396  *OptimumCoefVar = MaxCoefVar;
397 
398  if (optParams.Display)
399  {
400  UMUQMSG("Search on CoefVarFun : CoefVar(p)=", *OptimumCoefVar);
401  UMUQMSG("Converged to minimum at p = ", *OptimumP);
402  }
403  return true;
404  }
405  else
406  {
407  // If optimumP is not within Tolerance, we can go back and retry with better refinement (more iterations)
408 
409  MinValp = optimumP - 10 * stepSize;
410  if (MinValp < 0)
411  {
412  MinValp = double{};
413  }
414 
415  MaxValp = optimumP + 10 * stepSize;
416  if (MaxValp > double{4})
417  {
418  MaxValp = double{4};
419  }
420 
421  stepSize /= double{10};
422  if (stepSize < machinePrecision<double>)
423  {
424  // Fail:-1
425  UMUQFAILRETURN("Could not find the optimum value of p through search!");
426  }
427  }
428  }
429  UMUQFAILRETURN("Could not find the optimum value of p through search!");
430 }

432 bool tmcmcStats::selectNewGeneration(stdata &StreamData, database &CurrentData, runinfo &RunData, database &Leaders)
433 {
434  // current generation
435  int const currentGeneration = RunData.currentGeneration;
436 
437  // Next generation
438  int const newGeneration = currentGeneration + 1;
439 
440  // Probabilty at each generation
441  double *generationProbabilty = RunData.generationProbabilty.data();
442 
443  // Get the pointer to the function values
444  double *Fvalue = CurrentData.fValue.data();
445 
446  // Total number of function values
447  int const nCurrentSamplePoints = static_cast<int>(CurrentData.size());
448 
449  {
450  double optimumP(double{});
451  double optimumCoefVar(double{});
452 
453  // Probability of the current generation
454  double const PJ = generationProbabilty[currentGeneration];
455 
456  bool status(true);
457 
458  // Find the optimum value of \f$ p_{j+1} \f$ through optimization
459  if (!findOptimumP(Fvalue, nCurrentSamplePoints, PJ, &optimumP, &optimumCoefVar))
460  {
461  // In case we did not find the optimum value through optimization, we do the direct search
462  status = searchOptimumP(Fvalue, nCurrentSamplePoints, PJ, &optimumP, &optimumCoefVar);
463  }
464 
465  // If we find the optimum value a
466  if (status && optimumP > PJ)
467  {
468  generationProbabilty[newGeneration] = optimumP;
469  RunData.CoefVar[newGeneration] = optimumCoefVar;
470  }
471  else
472  {
473  // Increase the probability just a very small value
474  generationProbabilty[newGeneration] = PJ + 0.1 * optParams.Step;
475  RunData.CoefVar[newGeneration] = RunData.CoefVar[currentGeneration];
476  }
477  }
478 
479  // If the probability of this generation is greater than one, then it is the last step
480  if (generationProbabilty[newGeneration] > 1)
481  {
482  generationProbabilty[newGeneration] = double{1};
483  StreamData.eachPopulationSize[newGeneration] = StreamData.lastPopulationSize;
484  }
485 
486  std::vector<double> weight(nCurrentSamplePoints);
487 
488  {
489  // Get the PJ1 - PJ
490  double const probabiltyDiff = generationProbabilty[newGeneration] - generationProbabilty[currentGeneration];
491 
492  // The function value is in log space
493  for (int i = 0; i < nCurrentSamplePoints; i++)
494  {
495  weight[i] = Fvalue[i] * probabiltyDiff;
496  }
497  }
498 
499  stats s;
500  double const weightMax = s.maxelement(weight);
501 
502  std::for_each(weight.begin(), weight.end(), [&](double &w_i) { w_i = std::exp(w_i - weightMax); });
503 
504  if (optParams.Display)
505  {
506  io f;
507  f.printMatrix("Weight matrix", weight.data(), 1, nCurrentSamplePoints);
508  }
509 
510  // Compute the weight sum
511  double const weightSum = s.sum(weight);
512 
513  // Normalize the weight
514  std::for_each(weight.begin(), weight.end(), [&](double &w_i) { w_i /= weightSum; });
515 
516  if (optParams.Display)
517  {
518  io f;
519  f.printMatrix("Normalized Weight matrix", weight.data(), 1, nCurrentSamplePoints);
520  }
521 
522  {
523  // Keep the value for computing evidence
524  RunData.logselection[currentGeneration] = std::log(weightSum) + weightMax - std::log(static_cast<double>(nCurrentSamplePoints));
525 
526  if (optParams.Display)
527  {
528  io f;
529  f.printMatrix("log selection matrix", RunData.logselection.data(), 1, newGeneration);
530  }
531  }
532 
533  // Compute the current Generation CoefVar
534  RunData.CoefVar[currentGeneration] = s.coefvar(weight);
535 
536  if (optParams.Display)
537  {
538  io f;
539  f.printMatrix("CoefVar matrix", RunData.CoefVar.data(), 1, newGeneration);
540  }
541 
542  // Dimension of the sampling points
543  int const nDimSamplePoints = CurrentData.nDimSamplePoints;
544 
545  {
546  if (Leaders.nDimSamplePoints != nDimSamplePoints)
547  {
548  UMUQFAILRETURN("Sampling data dimension does not match with the leaders database!");
549  }
550 
551  if (Leaders.nSamplePoints < nCurrentSamplePoints)
552  {
553  UMUQFAILRETURN("Leader database size is not large enough!");
554  }
555 
556  // Get the total number of samples at the current generation
557  int const nSamples = StreamData.eachPopulationSize[currentGeneration];
558 
559  // Get the number of selection based on the probability
560  umuq::randomdist::multinomialDistribution<double> multinomial(weight.data(), nCurrentSamplePoints, nSamples);
561 
562  multinomial.dist(Leaders.nSelection);
563 
564  if (optParams.Display)
565  {
566  io f;
567  f.printMatrix("Number of selection = [", Leaders.nSelection, 1, nCurrentSamplePoints, "]\n----------------------------------------\n");
568  }
569  }
570 
571  // Total size of the sampling points array
572  int const nSize = nCurrentSamplePoints * nDimSamplePoints;
573 
574  // Compute vectors of mean and standard deviation for each dimension
575  std::vector<double> thetaMean(nDimSamplePoints);
576 
577  for (int i = 0; i < nDimSamplePoints; i++)
578  {
579  arrayWrapper<double> Parray(CurrentData.samplePoints.data() + i, nSize, nDimSamplePoints);
580  thetaMean[i] = std::inner_product(weight.begin(), weight.end(), Parray.begin(), double{});
581  }
582 
583  std::copy(thetaMean.begin(), thetaMean.end(), RunData.meantheta.data() + currentGeneration * nDimSamplePoints);
584 
585  if (optParams.Display)
586  {
587  io f;
588  f.printMatrix("Mean of theta = [", thetaMean.data(), 1, nDimSamplePoints, "]\n----------------------------------------\n");
589  }
590 
591  {
592  double *Covariance = RunData.covariance.data();
593 
594  for (int i = 0; i < nDimSamplePoints; i++)
595  {
596  arrayWrapper<double> iArray(CurrentData.samplePoints.data() + i, nSize, nDimSamplePoints);
597 
598  double const iMean = thetaMean[i];
599 
600  for (int j = i; j < nDimSamplePoints; j++)
601  {
602  arrayWrapper<double> jArray(CurrentData.samplePoints.data() + j, nSize, nDimSamplePoints);
603 
604  double const jMean = thetaMean[j];
605 
606  double S(0);
607  auto k = weight.begin();
608  for (auto i = iArray.begin(), j = jArray.begin(); i != iArray.end(); i++, j++, k++)
609  {
610  double const d1 = *i - iMean;
611  double const d2 = *j - jMean;
612  S += *k * d1 * d2;
613  }
614  Covariance[i * nDimSamplePoints + j] = S;
615  }
616  }
617 
618  for (int i = 0; i < nDimSamplePoints; i++)
619  {
620  for (int j = 0; j < i; j++)
621  {
622  Covariance[i * nDimSamplePoints + j] = Covariance[j * nDimSamplePoints + i];
623  }
624  }
625 
626  // If the covariance matrix is not positive definite
627  if (!umuq::linearalgebra::isSelfAdjointMatrixPositiveDefinite(Covariance, nDimSamplePoints))
628  {
629  // We force it to be positive definite
630  umuq::linearalgebra::forceSelfAdjointMatrixPositiveDefinite(Covariance, nDimSamplePoints);
631  }
632 
633  if (optParams.Display)
634  {
635  io f;
636  f.printMatrix("Covariance = [", Covariance, nDimSamplePoints, nDimSamplePoints, "]\n----------------------------------------\n");
637  }
638  }
639  return true;
640 }

642 double CoefVar(double const *FunValues, int const nFunValues, int const Stride, double const PJ1, double const PJ, double const Tolerance)
643 {
644  arrayWrapper<double> iArray(FunValues, nFunValues, Stride);
645 
646  std::vector<double> weight(iArray.size());
647 
648  stats s;
649 
650  {
651  //Find the maximum value in the array of FunValues of size nFunValues
652  double const fMaxValue = s.maxelement(iArray);
653 
654  double const diff = PJ1 - PJ;
655 
656  //Compute the weight
657  for (std::size_t i = 0; i < iArray.size(); i++)
658  {
659  weight[i] = std::exp((iArray[i] - fMaxValue) * diff);
660  }
661  }
662 
663  { //Compute the summation of weight
664  double const weightsum = s.sum(weight);
665 
666  //Normalize the weight
667  std::for_each(weight.begin(), weight.end(), [&](double &w) { w /= weightsum; });
668  }
669 
670  //Compute the mean
671  double const weightmean = s.mean(weight);
672 
673  //Compute the standard deviation
674  double const weightstddev = s.stddev(weight, weightmean);
675 
676  //return the square of the coefficient of variation (COV)
677  return std::pow(weightstddev / weightmean - Tolerance, 2);
