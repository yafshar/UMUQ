504 bool priorDistribution<RealType>::set(RealType const *Param1, RealType const *Param2, umuq::priorTypes const *compositeprior)
505 {
506  switch (priorType)
507  {
508  case umuq::priorTypes::UNIFORM:
509  try
510  {
511  uniformDist.reset(new umuq::density::uniformDistribution<RealType>(Param1, Param2, nDim * 2));
512  }
513  catch (...)
514  {
515  UMUQFAILRETURN("Failed to allocate memory!");
516  }
517  break;
518  case umuq::priorTypes::GAUSSIAN:
519  try
520  {
521  multivariateGaussianDist.reset(new umuq::density::multivariateGaussianDistribution<RealType>(Param1, Param2, nDim * 2));
522  }
523  catch (...)
524  {
525  UMUQFAILRETURN("Failed to allocate memory!");
526  }
527  break;
528  case umuq::priorTypes::EXPONENTIAL:
529  try
530  {
531  exponentialDist.reset(new umuq::density::exponentialDistribution<RealType>(Param1, nDim * 2));
532  }
533  catch (...)
534  {
535  UMUQFAILRETURN("Failed to allocate memory!");
536  }
537  break;
538  case umuq::priorTypes::GAMMA:
539  try
540  {
541  gammaDist.reset(new umuq::density::gammaDistribution<RealType>(Param1, Param2, nDim * 2));
542  }
543  catch (...)
544  {
545  UMUQFAILRETURN("Failed to allocate memory!");
546  }
547  break;
548  case umuq::priorTypes::COMPOSITE:
549  {
550  if (compositeprior)
551  {
552  try
553  {
554  compositePrior.reset(new umuq::priorTypes[nDim]());
555  }
556  catch (...)
557  {
558  UMUQFAILRETURN("Failed to allocate memory!");
559  }
560 
561  std::copy(compositeprior, compositeprior + nDim, compositePrior.get());
562  }
563  else
564  {
565  UMUQFAILRETURN("Failed to provide composite prior types for each dimension!");
566  }
567 
568  if (multivariateGaussianDist)
569  {
570  multivariateGaussianDist.reset(nullptr);
571  }
572 
573  int nUNIFORM(0);
574  int nGAUSSIAN(0);
575  int nEXPONENTIAL(0);
576  int nGAMMA(0);
577 
578  for (int i = 0; i < nDim; i++)
579  {
580  switch (compositePrior[i])
581  {
582  case umuq::priorTypes::UNIFORM:
583  nUNIFORM++;
584  break;
585  case umuq::priorTypes::GAUSSIAN:
586  nGAUSSIAN++;
587  break;
588  case umuq::priorTypes::EXPONENTIAL:
589  nEXPONENTIAL++;
590  break;
591  case umuq::priorTypes::GAMMA:
592  nGAMMA++;
593  break;
594  default:
595  UMUQFAILRETURN("Unknown prior distribution type!");
596  break;
597  };
598  }
599 
600  if (nUNIFORM)
601  {
602  uniformIndex.resize(nUNIFORM);
603  uniformPoints.resize(nUNIFORM);
604  }
605  if (nGAUSSIAN)
606  {
607  gaussianIndex.resize(nGAUSSIAN);
608  gaussianPoints.resize(nGAUSSIAN);
609  }
610  if (nEXPONENTIAL)
611  {
612  exponentialIndex.resize(nEXPONENTIAL);
613  exponentialPoints.resize(nEXPONENTIAL);
614  }
615  if (nGAMMA)
616  {
617  gammaIndex.resize(nGAMMA);
618  gammaPoints.resize(nGAMMA);
619  }
620 
621  std::vector<RealType> uparam1(nUNIFORM);
622  std::vector<RealType> uparam2(nUNIFORM);
623  std::vector<RealType> nparam1(nGAUSSIAN);
624  std::vector<RealType> nparam2(nGAUSSIAN);
625  std::vector<RealType> eparam1(nEXPONENTIAL);
626  std::vector<RealType> gparam1(nGAMMA);
627  std::vector<RealType> gparam2(nGAMMA);
628 
629  nUNIFORM = 0;
630  nGAUSSIAN = 0;
631  nEXPONENTIAL = 0;
632  nGAMMA = 0;
633 
634  for (int i = 0; i < nDim; i++)
635  {
636  switch (compositePrior[i])
637  {
638  case umuq::priorTypes::UNIFORM:
639  uparam1[nUNIFORM] = Param1[i];
640  uparam2[nUNIFORM] = Param2[i];
641  uniformIndex[nUNIFORM] = i;
642  nUNIFORM++;
643  break;
644  case umuq::priorTypes::GAUSSIAN:
645  nparam1[nGAUSSIAN] = Param1[i];
646  nparam2[nGAUSSIAN] = Param2[i];
647  gaussianIndex[nGAUSSIAN] = i;
648  nGAUSSIAN++;
649  break;
650  case umuq::priorTypes::EXPONENTIAL:
651  eparam1[nEXPONENTIAL] = Param1[i];
652  exponentialIndex[nEXPONENTIAL] = i;
653  nEXPONENTIAL++;
654  break;
655  case umuq::priorTypes::GAMMA:
656  gparam1[nGAMMA] = Param1[i];
657  gparam2[nGAMMA] = Param2[i];
658  gammaIndex[nGAMMA] = i;
659  nGAMMA++;
660  break;
661  default:
662  UMUQFAILRETURN("Unknown prior distribution type!");
663  break;
664  };
665  }
666 
667  if (nUNIFORM)
668  {
669  try
670  {
671  uniformDist.reset(new umuq::density::uniformDistribution<RealType>(uparam1.data(), uparam2.data(), nUNIFORM * 2));
672  }
673  catch (...)
674  {
675  UMUQFAILRETURN("Failed to allocate memory!");
676  }
677  }
678  else
679  {
680  if (uniformDist)
681  {
682  uniformDist.reset(nullptr);
683  }
684  }
685  if (nGAUSSIAN)
686  {
687  try
688  {
689  gaussianDist.reset(new umuq::density::gaussianDistribution<RealType>(nparam1.data(), nparam2.data(), nGAUSSIAN * 2));
690  }
691  catch (...)
692  {
693  UMUQFAILRETURN("Failed to allocate memory!");
694  }
695  }
696  else
697  {
698  gaussianDist.reset(nullptr);
699  }
700  if (nEXPONENTIAL)
701  {
702  try
703  {
704  exponentialDist.reset(new umuq::density::exponentialDistribution<RealType>(eparam1.data(), nEXPONENTIAL));
705  }
706  catch (...)
707  {
708  UMUQFAILRETURN("Failed to allocate memory!");
709  }
710  }
711  else
712  {
713  if (exponentialDist)
714  {
715  exponentialDist.reset(nullptr);
716  }
717  }
718  if (nGAMMA)
719  {
720  try
721  {
722  gammaDist.reset(new umuq::density::gammaDistribution<RealType>(gparam1.data(), gparam2.data(), nGAMMA * 2));
723  }
724  catch (...)
725  {
726  UMUQFAILRETURN("Failed to allocate memory!");
727  }
728  }
729  else
730  {
731  if (gammaDist)
732  {
733  gammaDist.reset(nullptr);
734  }
735  }
736  }
737 
738  break;
739  default:
740  UMUQFAILRETURN("Unknown prior distribution type!");
741  break;
742  };
743 
744  return true;

789 template <typename RealType>
790 RealType priorDistribution<RealType>::pdf(RealType const *x)
791 {
792  switch (priorType)
793  {
794  case umuq::priorTypes::UNIFORM:
795  return uniformDist->f(x);
796  break;
797  case umuq::priorTypes::GAUSSIAN:
798  return multivariateGaussianDist->f(x);
799  break;
800  case umuq::priorTypes::EXPONENTIAL:
801  return exponentialDist->f(x);
802  break;
803  case umuq::priorTypes::GAMMA:
804  return gammaDist->f(x);
805  break;
806  case umuq::priorTypes::COMPOSITE:
807  RealType sum(1);
808  if (uniformDist)
809  {
810  int j(0);
811  for (auto i : uniformIndex)
812  {
813  uniformPoints[j++] = x[i];
814  }
815  sum *= uniformDist->f(uniformPoints.data());
816  }
817  if (gaussianDist)
818  {
819  int j(0);
820  for (auto i : gaussianIndex)
821  {
822  gaussianPoints[j++] = x[i];
823  }
824  sum *= gaussianDist->f(gaussianPoints.data());
825  }
826  if (exponentialDist)
827  {
828  int j(0);
829  for (auto i : exponentialIndex)
830  {
831  exponentialPoints[j++] = x[i];
832  }
833  sum *= exponentialDist->f(exponentialPoints.data());
834  }
835  if (gammaDist)
836  {
837  int j(0);
838  for (auto i : gammaIndex)
839  {
840  gammaPoints[j++] = x[i];
841  }
842  sum *= gammaDist->f(gammaPoints.data());
843  }
844  return sum;
845  break;
846  }
847  UMUQFAIL("Unknown Prior type!");
848 }
