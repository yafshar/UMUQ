435 void psrandom::initTask()
436 {
437  std::vector<std::size_t> rSeed(std::mt19937::state_size);
438 
439  // Get the local number of workers
440  std::size_t const nlocalworkers = static_cast<std::size_t>(torc_i_num_workers());
441 
442  // Node Id (MPI rank)
443  std::size_t const node_id = static_cast<std::size_t>(torc_node_id());
444 
445  std::size_t const n = nlocalworkers * (node_id + 1);
446 
447  rSeed[0] = PRNG_seed;
448  rSeed[1] = n;
449 
450  for (std::size_t i = 0; i < nlocalworkers; i++)
451  {
452  std::iota(rSeed.begin() + 2, rSeed.end(), i);
453 
454  // Seed the engine with unsigned ints
455  std::seed_seq sSeq(rSeed.begin(), rSeed.end());
456 
457  // For each thread feed the RNG
458  NumberGenerator[i].seed(sSeq);
459 
460  saru[i].seed(rSeed[0], rSeed[1], rSeed[2]);
461  }
462 }
