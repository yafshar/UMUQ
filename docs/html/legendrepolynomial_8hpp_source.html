405 template <typename RealType>
406 RealType LegendrePolynomial<RealType>::legendre(int const n, RealType const x, bool const second)
407 {
408  // Error handling:
409  if ((x < -1) || (x > 1))
410  {
411  UMUQFAIL("The Legendre Polynomial is defined for -1 <= x <= 1, but got x =", x);
412  }
413 
414  RealType P0;
415  RealType P1;
416  if (second)
417  {
418  // A solution of the second kind (Q):
419  P0 = (std::log1p(x) - std::log1p(-x)) / static_cast<RealType>(2);
420  P1 = x * P0 - static_cast<RealType>(1);
421  }
422  else
423  {
424  // A solution of the first kind (P):
425  P0 = static_cast<RealType>(1);
426  P1 = x;
427  }
428  if (n == 0)
429  {
430  return P0;
431  }
432 
433  int i(1);
434  while (i < n)
435  {
436  std::swap(P0, P1);
437  P1 = this->legendre_next(i, x, P0, P1);
438  ++i;
439  }
440  return P1;
441 }

444 RealType *LegendrePolynomial<RealType>::legendre_array(int const n, RealType const x, bool const second)
445 {
446  // Error handling:
447  if ((x < -1) || (x > 1))
448  {
449  UMUQFAIL("The Legendre Polynomial is defined for -1 <= x <= 1, but got x =", x);
450  }
451 
452  RealType *results = nullptr;
453 
454  try
455  {
456  results = new RealType[n + 1];
457  }
458  catch (...)
459  {
460  UMUQFAIL("Failed to allocate memory!");
461  }
462 
463  RealType P0;
464  RealType P1;
465  if (second)
466  {
467  // A solution of the second kind (Q):
468  P0 = (std::log1p(x) - std::log1p(-x)) / static_cast<RealType>(2);
469  P1 = x * P0 - static_cast<RealType>(1);
470  }
471  else
472  {
473  // A solution of the first kind (P):
474  P0 = static_cast<RealType>(1);
475  P1 = x;
476  }
477 
478  results[0] = P0;
479  if (n == 0)
480  {
481  return results;
482  }
483 
484  results[1] = P1;
485 
486  int i(1);
487  while (i < n)
488  {
489  std::swap(P0, P1);
490  P1 = this->legendre_next(i, x, P0, P1);
491  ++i;
492  results[i] = P1;
493  }
