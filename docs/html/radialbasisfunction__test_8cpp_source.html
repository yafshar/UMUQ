45 TEST(radialBasisFunction_test, HandlesFunctions)

51  umuq::EMatrixXd Points = (umuq::EMatrixXd::Random(nDimensions, maxNumPoints) + umuq::EMatrixXd::Ones(nDimensions, maxNumPoints)) / 2;

53 
54  umuq::EMatrixXd QPoints = (umuq::EMatrixXd::Random(nDimensions, nQueryPoints) + umuq::EMatrixXd::Ones(nDimensions, nQueryPoints)) / 2;
55  umuq::EVectorXd QFunctionValues = (QPoints.row(1).array() * QPoints.row(0).array().sin() + QPoints.row(0).array() * QPoints.row(1).array().cos()).matrix().transpose();
56 
57  umuq::radialBasisFunction<umuq::cubicKernel, umuq::linearPolynomialTail> RBF(nDimensions, maxNumPoints, 0);
58 
59  EXPECT_TRUE(RBF.addPoint(Points, FunctionValues));
60 
61  EXPECT_TRUE(RBF.fit());
62 
63  // Evaluate at the center to see that we are interpolating
64  {
65  umuq::EVectorXd PointsEvaluation = RBF.evaluate(Points);
66  EXPECT_FALSE((PointsEvaluation - FunctionValues).cwiseAbs().maxCoeff() > 1e-10);
67  }
68 
69  // Evaluate at some other points
70  {
71  umuq::EVectorXd PointsEvaluation = RBF.evaluate(QPoints);
72  EXPECT_FALSE((PointsEvaluation - QFunctionValues).cwiseAbs().maxCoeff() > 1e-3);
73 
74  umuq::EMatrixXd Distance = umuq::L2Distance<double>(QPoints, Points);
75  EXPECT_FALSE((PointsEvaluation - RBF.evaluate(QPoints, Distance)).norm() > 1e-10);
76  }
77 
78  // Look at derivatives
79  {
80  umuq::EMatrixXd DerivativeValues(nDimensions, nQueryPoints);
81  DerivativeValues.row(0) = (QPoints.row(1).array() * QPoints.row(0).array().cos() + QPoints.row(1).array().cos()).matrix();
82  DerivativeValues.row(1) = (QPoints.row(0).array().sin() - QPoints.row(0).array() * QPoints.row(1).array().sin()).matrix();
83 
84  for (auto i = 0; i < nQueryPoints; ++i)
85  {
86  umuq::EVectorXd Derivative = RBF.deriv(QPoints.col(i));
87  EXPECT_FALSE((DerivativeValues.col(i) - Derivative).norm() >= 1e-2);
88  }
89  }
90 
91  // Reset the Object
92  RBF.reset();
93 
94  // Add all points again
95  EXPECT_TRUE(RBF.addPoint(Points.block(0, 0, nDimensions, maxNumPoints - 10), FunctionValues.head(maxNumPoints - 10)));
96  EXPECT_TRUE(RBF.addPoint(Points.block(0, maxNumPoints - 10, nDimensions, 5), FunctionValues.segment(maxNumPoints - 10, 5)));
97  for (int i = maxNumPoints - 5; i < maxNumPoints; i++)
98  {
99  EXPECT_TRUE(RBF.addPoint(Points.col(i), FunctionValues(i)));
100  }
101  EXPECT_FALSE(RBF.addPoint(Points.col(0), FunctionValues(0)));
102  EXPECT_TRUE(RBF.fit());
103 }
