251 template <typename RealType, class FunctionType>
252 inline RealType multivariateGaussianDistribution<RealType, FunctionType>::multivariategaussianDistribution_f(RealType const *x)
253 {
254  RealType const denom = std::pow(static_cast<RealType>(M_2PI), this->numParams) * mvnormal->lu.determinant();
255  EVectorX<RealType> ax = EVectorMapTypeConst<RealType>(x, this->numParams) - mvnormal->mean;
256  // Mahalanobis distance between \f$ X \f$ and \f$ \mu \f$
257  RealType const MDistSq = ax.transpose() * mvnormal->lu.inverse() * ax;
258  return std::exp(-0.5 * MDistSq) / std::sqrt(denom);
259 }

261 template <typename RealType, class FunctionType>
262 inline RealType multivariateGaussianDistribution<RealType, FunctionType>::multivariategaussianDistribution_lf(RealType const *x)
263 {
264  EVectorX<RealType> ax = EVectorMapTypeConst<RealType>(x, this->numParams) - mvnormal->mean;
265  // Mahalanobis distance between \f$ X \f$ and \f$ \mu \f$
266  RealType const MDistSq = ax.transpose() * mvnormal->lu.inverse() * ax;
267  return -0.5 * (MDistSq + this->numParams * static_cast<RealType>(M_L2PI) + std::log(mvnormal->lu.determinant()));
268 }
