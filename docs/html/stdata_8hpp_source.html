569  if (f.isFileExist(fname))

572  {
573  // We need a parser object to parse

576  // These are temporary variables
577  int probdim = nDim;
578  int maxgens = maxGenerations;

582  while (f.readLine())
583  {
584  // Parse the line into line arguments
585  p.parse(f.getLine());
586 
587  if (p.at<std::string>(0) == "nDim")
588  {
589  nDim = p.at<int>(1);
590  }
591  else if (p.at<std::string>(0) == "maxGenerations")
592  {
593  maxGenerations = p.at<int>(1);
594  }
595  else if (p.at<std::string>(0) == "populationSize")
596  {
597  populationSize = p.at<int>(1);
598  }
599  }
600 
601  {
602  bool isInitialized = !(probdim == nDim && maxgens == maxGenerations && datanum == populationSize);
603  if (isInitialized)
604  {
605  if (!reset(nDim, maxGenerations, populationSize))
606  {
607  return false;
608  }
609  }
610  }
611 
612  f.rewindFile();
613 
614  // Read each line in the file and skip all the commented and empty line with the default comment "#"
615  while (f.readLine())
616  {
617  // Parse the line into line arguments
618  p.parse(f.getLine());
619 
620  if (p.at<std::string>(0) == "coefVarPresetThreshold")
621  {
622  coefVarPresetThreshold = p.at<double>(1);
623  }
624  else if (p.at<std::string>(0) == "bbeta")
625  {
626  bbeta = p.at<double>(1);
627  }
628  else if (p.at<std::string>(0) == "seed")
629  {
630  seed = p.at<long>(1);
631  }
632  else if (p.at<std::string>(0) == "opt.MaxIter")
633  {
634  options.MaxIter = p.at<int>(1);
635  }
636  else if (p.at<std::string>(0) == "opt.Tol")
637  {
638  options.Tolerance = p.at<double>(1);
639  }
640  else if (p.at<std::string>(0) == "opt.FMin" || p.at<std::string>(0) == "opt.Minimizer")
641  {
642  options.FunctionMinimizerType = p.at<double>(1);
643  }
644  else if (p.at<std::string>(0) == "opt.Display")
645  {
646  options.Display = p.at<int>(1);
647  }
648  else if (p.at<std::string>(0) == "opt.Step")
649  {
650  options.Step = p.at<double>(1);
651  }
652  else if (p.at<std::string>(0) == "priorType")
653  {
654 
655  priorType = static_cast<priorTypes>(p.at<int>(1));
656  }
657  else if (p.at<std::string>(0) == "iPlot")
658  {

660  }
661  else if (p.at<std::string>(0) == "saveData")
662  {
663  saveData = p.at<int>(1);
664  }
665  else if (p.at<std::string>(0) == "Bdef")

669  }

671  {
672  minChainLength = p.at<int>(1);
673  }

675  {
676  maxChainLength = p.at<int>(1);

679  {
680  useLocalCovariance = p.at<int>(1);
681  }
682  }
683 
684  int n = nDim;
685  int found;
686  while (n--)
687  {
688  f.rewindFile();
689 
690  found = 0;
691  std::string strTemp("B" + std::to_string(n));
692 
693  while (f.readLine())
694  {
695  p.parse(f.getLine());
696 
697  if (p.at<std::string>(0) == strTemp)
698  {
699  lowerBound[n] = p.at<double>(1);
700  upperBound[n] = p.at<double>(2);
701  found = 1;
702  break;
703  }
704  }
705 
706  // In case we do not find the value, we use the default lower bound and upper bound
707  if (!found)
708  {
709  lowerBound[n] = lb;
710  upperBound[n] = ub;
711  }
712  }
713 
714  // 0: uniform
715  if (priorType == priorTypes::UNIFORM)
716  {
717  for (n = 0; n < nDim; n++)
718  {

722  }
723

741  }
742 
743  f.rewindFile();
744

746  {
747  p.parse(f.getLine());
748 
749  if (p.at<std::string>(0) == "priorSigma")
750  {
751  for (n = 0; n < nDim * nDim; n++)
752  {
753  priorParam2[n] = p.at<double>(n + 1);
754  }
755  break;
756  }
757  }
758  }
759 
760  // 2: exponential
761  if (priorType == priorTypes::EXPONENTIAL)
762  {
763  f.rewindFile();
764 
765  while (f.readLine())
766  {
767  p.parse(f.getLine());
768
