295 template <typename T>
296 inline T parser::parse(const char *inputLineArg, T &parsedValue)
297 {
298  if (!inputLineArg)
299  {
300  UMUQFAIL("NULL string!");
301  }
302  if (umuq_float<T>::value)
303  {
304  int const n = std::strlen(std::strtok(const_cast<char *>(inputLineArg), " \t\n\r:,"));
305  if (n == 0)
306  {
307  UMUQFAIL("Expected floating type parameter instead of empty string!");
308  }
309  for (int i = 0; i < n; ++i)
310  {
311  if (std::isdigit(inputLineArg[i]) || inputLineArg[i] == '-' || inputLineArg[i] == '+')
312  continue;
313  if (inputLineArg[i] == '.' || inputLineArg[i] == 'e' || inputLineArg[i] == 'E')
314  continue;
315  std::string msg("Expected floating point parameter instead of '");
316  for (int j = 0; j < n; ++j)
317  msg += inputLineArg[j];
318  msg += "'";
319  UMUQFAIL(msg);
320  }
321  }
322  if (umuq_int<T>::value)
323  {
324  int const n = std::strlen(std::strtok(const_cast<char *>(inputLineArg), " \t\n\r:,"));
325  if (n == 0)
326  {
327  UMUQFAIL("Expected integer type parameter instead of empty string!");
328  }
329  for (int i = 0; i < n; ++i)
330  {
331  if (std::isdigit(inputLineArg[i]) || inputLineArg[i] == '-' || inputLineArg[i] == '+')
332  continue;
333  std::string msg("Expected integer parameter instead of '");
334  for (int j = 0; j < n; ++j)
335  msg += inputLineArg[j];
336  msg += "'";
337  UMUQFAIL(msg);
338  }
339  }
340  std::stringstream str(inputLineArg);
341  str >> parsedValue;
342  return parsedValue;
343 }
